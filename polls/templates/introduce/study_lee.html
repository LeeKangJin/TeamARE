def generatePointCluster(clusterNumber, pointNumber):  # pointNumber * clusterNumber 만큼의 데이터를 생성한다. 
    randomCordinateList = []

    for j in range(clusterNumber):
        for i in range(pointNumber):
            randomNumberX = r.randrange(1, 100) / 100
            randomNumberY = r.randrange(1, 100) / 100
            randomCordinateList.append([randomNumberX, randomNumberY])

    return randomCordinateList  # (x,y) list

def computeDistance(Acordinate, Bcordinate):  # root ((x1-x2)^2 + (x1-x2)^2)

    # x좌표 계산
    xdistance = computeAbsDistance(Acordinate[0], Bcordinate[0])
    ydistance = computeAbsDistance(Acordinate[1], Bcordinate[1])
    return math.sqrt(xdistance * xdistance + ydistance * ydistance)

def computeAbsDistance(pointA, pointB):     #|(Xa - Xa)|
    if (pointA < pointB):
        return pointB - pointA
    elif (pointA > pointB):
        return pointA - pointB
    else:
        return 0

def average(list):           #평균값 구하기
    x = 0
    for i in range(len(list)):
        x = x + list[i]

    return x / len(list)

def computeCenter(clusterNumber, cordinateList, labelList):          # center 구하기
    center = [[0 for j in range(2)] for i in range(clusterNumber)]   #[[0,0],[0,0] ...] 형식 초기화
    dataLen = len(cordinateList)

    x = [[] for i in range(clusterNumber)]
    y = [[] for i in range(clusterNumber)] # [[].[].[] ..] 형식으로 초기화

    for i in range(dataLen):

        for j in range(clusterNumber):

            if (labelList[i] == j):  # 라벨에 따라서 x,y 들을 집어넣음
                
                x[j].append(cordinateList[i][0])           #라벨이 j인경우 x[j]번째 리스트에 입력
                y[j].append(cordinateList[i][1])

    for k in range(clusterNumber):
       
        center[k][0] = average(x[k])                       #x[k] 
        center[k][1] = average(y[k])

    return center

def labelchange(clusterNumber, cordinateList, center, labelList):
    distanceList = []
    totaldistanceList = []
    for i in range(len(cordinateList)):

        for j in range(clusterNumber):
            distanceList.append(computeDistance(center[j], cordinateList[i])) #center와 점들간의 거리를 구함.

        totaldistanceList.append(min(distanceList))                           # 거리가 작은 값들을 totaldistance list에 넣음.
        labelList[i] = distanceList.index(min(distanceList))                  # 거리가 가장 작은 값에 해당하는 라벨을 갱신함.
        distanceList = []

    return labelList,totaldistanceList

def computeEnergy(dataList):
    # sum(center - dataList 에서 min)^2
    energy = 0
    for i in  range(len(dataList)):
        energy = energy + dataList[i]*dataList[i]

    return energy

def printData(cordinateList, labelList, center):
    x = [[], [], []]
    y = [[], [], []]

    for i in range(len(cordinateList)):
        for j in range(3):

            if (labelList[i] == j):
                x[j].append(cordinateList[i][0])
                y[j].append(cordinateList[i][1])

    plt.scatter(x[0], y[0], label="samples", color='r', marker='o')
    plt.scatter(x[1], y[1], label="samples", color='g', marker='o')
    plt.scatter(x[2], y[2], label="samples", color='b', marker='o')

    for i in range(len(center)):
        plt.scatter(center[i][0],center[i][1], label="samples", color='black',marker='o')

    plt.show()
    return 0
